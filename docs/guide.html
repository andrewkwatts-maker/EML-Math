Getting Started Guide

1. Installation

pip install eml-math         # core library
pip install eml-math[ext]    # + NumPy and SymPy
Pre-built wheels are on PyPI for Python 3.10–3.13 on Windows, macOS, and Linux. The Rust backend is included in the wheel — you do not need Rust installed.

2. Hello, EML

from eml_math import EMLPoint

# The single operator
print(EMLPoint(1, 1).eml())    # e  = 2.71828...
print(EMLPoint(2, 1).eml())    # exp(2) = 7.38905...

# All 36 elementary functions via operators.py
import eml_math.operators as ops
import math

print(ops.sin(math.pi / 6).tension())   # 0.5
print(ops.ln(math.e).tension())           # 1.0
print(ops.add(2, 3).tension())            # 5.0
print(ops.mul(4, 7).tension())            # 28.0

3. Mirror-Pulse Dynamics

from eml_math import EMLState, EMLPoint, simulate_pulses, verify_conservation

# Create an initial state
state = EMLState(EMLPoint(1.0, 1.0))

# Run 20 Mirror-Pulses
trajectory = simulate_pulses(state, n_pulses=20)

# Conservation holds at every step (Axiom 10)
print(verify_conservation(trajectory))   # True

# Inspect the trajectory
for s in trajectory[:5]:
    print(f"n={s.flip_count:2d}  rho={s.rho:.6f}  phase={s.phase:.4f}")

4. Complex Arithmetic Without Complex Numbers

from eml_math import EMLPair
import math

# Build from real and imaginary parts
z = EMLPair.from_values(3.0, 4.0)
print(z.modulus)     # 5.0
print(z.argument)    # arctan(4/3) ≈ 0.9273

# Arithmetic
z2 = EMLPair.from_values(1.0, -1.0)
print((z * z2).real_tension)   # 3*1 - 4*(-1) = 7
print((z + z2).real_tension)   # 4.0

# Polar construction
z3 = EMLPair.from_polar(r=2.0, theta=math.pi/4)
print(z3.real_tension)   # 2*cos(π/4) ≈ 1.414

5. Relativistic Geometry

from eml_math import EMLPoint
import math

p = EMLPoint(1.0, math.e)

# Causal classification
print(p.light_cone_type())      # "lightlike"
print(p.minkowski_delta())      # 0.0

# Create a timelike point
p2 = EMLPoint(2.0, math.e)
print(p2.is_timelike())         # True
print(p2.rapidity())            # atanh(1/e²)

# Lorentz boost — Minkowski delta is invariant
p3 = p2.boost(phi=0.693, c=1.0)
print(abs(p2.minkowski_delta() - p3.minkowski_delta()) < 1e-10)  # True

# 4-frame (canonical) coordinates
frame0 = p2.canonical_frame(k=0)   # (exp(x), ln(y))
frame1 = p2.canonical_frame(k=1)   # (-ln(y), exp(x))
# All frames have the same Euclidean delta

6. Spacetime Metrics and Geodesics

from eml_math.metric import MetricTensor
from eml_math import EMLPoint, EMLState

# Schwarzschild metric (r = exp(x), rs = Schwarzschild radius)
metric = MetricTensor.schwarzschild(rs=2.0)
p = EMLPoint(3.0, 1.0)   # r = exp(3) ≈ 20 >> rs=2

# Christoffel symbol Γ^0_{01} (Rust-accelerated for Schwarzschild)
print(metric.christoffel(0, 0, 1, p))

# Geodesic integration (Euler step)
state = EMLState.from_point(p)
for _ in range(50):
    state = state.geodesic_step(metric, dtau=0.005)

# FLRW cosmology: radiation-era scale factor a(t) ∝ t^0.5
flrw = MetricTensor.flrw(scale_factor_a=lambda t: t**0.5)
g = flrw._g(EMLPoint(0.5, 2.0))
print(g)   # [[-1, 0], [0, a(t)²]]

7. Four-Momentum

from eml_math.momentum import FourMomentum

# From rest mass + velocity
fm = FourMomentum.from_mass_velocity(mass=1.0, v=0.8, c=1.0)
print(f"E={fm.energy:.4f}  p={fm.momentum:.4f}  m={fm.mass:.4f}  γ={fm.gamma():.4f}")
# Mass is Lorentz invariant
fm2 = fm.boost(phi=0.5)
print(abs(fm.mass - fm2.mass) < 1e-10)   # True

8. Octonions

from eml_math.octonion import basis_octonion, is_g2_automorphism

e1, e2, e4 = basis_octonion(1), basis_octonion(2), basis_octonion(4)

# Fano-plane rule: e1 * e2 = e4
result = e1 * e2
print(result.component(4))   # 1.0

# Non-commutativity: e2 * e1 = -e4
print((e2 * e1).component(4))   # -1.0

# Norm is multiplicative
print(abs((e1 * e2).norm() - e1.norm() * e2.norm()) < 1e-9)   # True

# Check G₂ automorphism
print(is_g2_automorphism(e1, e2, lambda o: o))   # True (identity)

9. Clifford Algebra

from eml_math.geometric_algebra import EMLMultivector
from eml_math import EMLPoint
import math

# 2D Euclidean: signature (1, 1)
def vec2(a, b, sig=(1, 1)):
    comps = [EMLPoint(0.0, 1.0)] * 4
    comps[1] = EMLPoint(a, 1.0)
    comps[2] = EMLPoint(b, 1.0)
    return EMLMultivector(comps, signature=sig)

v = vec2(3.0, 4.0)
print(v.quadratic())   # 9 + 16 = 25

# Rotor: rotate by π/4 in the e0∧e1 plane
R = v.rotor(math.pi / 4, plane=(0, 1))
v_rot = v.rotate(R)
print(abs(v_rot.quadratic() - v.quadratic()) < 1e-8)   # True

# Spacetime multivector Cl(1,3)
st = EMLMultivector.spacetime([EMLPoint(0.0, 1.0)] * 16)

10. Using the Rust Batch API Directly

from eml_math import eml_core
import math

# Batch boost 10k points in parallel (Rayon)
N = 10_000
points = [(1.0, math.e)] * N
phis   = [0.5] * N
boosted = eml_core.boost_n(points, phis, c=1.0)

# Batch Schwarzschild Christoffel
results = eml_core.christoffel_batch_n(points, 0, 0, 1, rs=2.0)

# Direct Rust EMLPoint
rp = eml_core.EMLPoint(1.0, math.e)
print(rp.euclidean_delta())
print(rp.boost(0.5, 1.0))

11. C / C++ Integration

// Build the C library first:
// cargo build --release -p eml_c_api

// In your C program:
#include "eml_math.h"
#include <stdio.h>

int main(void) {
    double t = eml_tension(1.0, 1.0);   // e
    printf("eml(1,1) = %f\n", t);

    double ox, oy;
    eml_boost(1.0, 2.718, 0.5, 1.0, &ox, &oy);

    double e1[8] = {0,1,0,0,0,0,0,0};
    double e2[8] = {0,0,1,0,0,0,0,0};
    double res[8];
    eml_octonion_mul(e1, e2, res);   // res[4] == 1.0
    return 0;
}
# Compile
gcc main.c -Ic_api -Ltarget/release -leml_math -lm -o main
g++ main.cpp -Ic_api -Ltarget/release -leml_math -lm -o main  # C++ works too
